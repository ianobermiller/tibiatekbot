LevelSpy(non-debug) & Reveal Invisible - 12-21-2007, 11:00 AM
I'm to lazy to make a real tutorial so I'll just post an example on how to use Level Spy and Reveal Invisible.
Code:
/**********************************
* Level spy example for Tibia 8.1 *
* By Stiju                        *
**********************************/
#include <windows.h>

#define LEVELSPY1			0x4E115A
#define LEVELSPY2			0x4E125F
#define LEVELSPY3			0x4E12E0
#define LEVELSPYPTR			0x61B608
BYTE NOP[6]				= {0x90, 0x90, 0x90, 0x90, 0x90, 0x90};
BYTE ORIGINAL[6]			= {0x89, 0x86, 0xD8, 0x25, 0x00, 0x00};

int main()
{
	// Find Tibia window and get the process handle
	HWND hWnd = FindWindow("TibiaClient", 0);
	DWORD dwProcessId;
	GetWindowThreadProcessId(hWnd, &dwProcessId);
	HANDLE hProcess = OpenProcess(PROCESS_ALL_ACCESS, false, dwProcessId);

	// NOP 6 bytes in the 3 first LEVELSPY addresses 
	WriteProcessMemory(hProcess, (LPVOID)LEVELSPY1, &NOP, 6, 0);
	WriteProcessMemory(hProcess, (LPVOID)LEVELSPY2, &NOP, 6, 0);
	WriteProcessMemory(hProcess, (LPVOID)LEVELSPY3, &NOP, 6, 0);

	// Read LEVELSPYPTR and save value in dwTemp, add 28 to dwTemp.
	// Then read dwTemp, save the value in dwLevelSpy and add 0x25D8 to it.
	// dwLevelSpy will hold the address you should change later to change floor.
	DWORD dwLevelSpy, dwTemp;
	ReadProcessMemory(hProcess, (LPVOID)LEVELSPYPTR, &dwTemp, 4, 0);
	dwTemp += 28;
	ReadProcessMemory(hProcess, (LPVOID)dwTemp, &dwLevelSpy, 4, 0);
	dwLevelSpy += 0x25D8;

	// Loop through all seven floors.
	// This will also work underground without debugs.
	for(int nFloor = 0; nFloor < 7; nFloor++)
	{
		// Write nFloor value to dwLevelSpy address.
		WriteProcessMemory(hProcess, (LPVOID)dwLevelSpy, &nFloor, 4, 0);
		Sleep(1000);
	}

	// Change back the 3 LEVELSPY addresses to the original values.
	WriteProcessMemory(hProcess, (LPVOID)LEVELSPY3, &ORIGINAL, 6, 0);
	WriteProcessMemory(hProcess, (LPVOID)LEVELSPY2, &ORIGINAL, 6, 0);
	WriteProcessMemory(hProcess, (LPVOID)LEVELSPY1, &ORIGINAL, 6, 0);
	return 0;
}
To make invisible creatures looks like invisible players.
All you have to do is change 1 byte from these addresses:
Code:
#define INVISIBLE1 0x00453AF3 (Change from 0x72 to 0xEB) // JB SHORT 00453AFF
#define INVISIBLE2 0x004DE734 (Change from 0x75 to 0xEB) // JNZ SHORT 004DE73F
If you want to patch your Tibia.exe with this you should use the address 0x53AF3 and 0xDE734.
